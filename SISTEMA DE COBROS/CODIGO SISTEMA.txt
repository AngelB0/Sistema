CREATE TABLE ALUMNOS(
	matricula INT PRIMARY KEY,
	alumno varchar(70),
	plantel varchar(50),
	oferta_edu varchar(50),
	grado INT,
	grupo varchar(15),
	periodo varchar(50),
	turno varchar(15),
	fecha_ins date,
	estatus varchar(15),
	fecha_est date
);

CREATE TABLE VENTAS(
	folio INT NOT NULL PRIMARY KEY,
	plantel varchar(50),
	matri INT,
	nombre varchar(70),
	grupo varchar(15),
	licenciatura varchar(50),
	fecha date,
	cant_recibe float,
	forma_p varchar(15),
	aprov varchar(15),
	cuenta_rec varchar(30),
	concep INT,
	observaciones varchar(250),
	subtotal float
);
